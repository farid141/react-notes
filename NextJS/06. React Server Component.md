# React Server Component (RSC)

Merupakan sebuah fitur NextJS untuk rendering document tree di server (client menerima berupa data HTML secara langsung). Berbeda dengan pembuatan ReactJS dengan vite yang merupakan Client Side Rendering (CSR).

dibagi menjadi 2:

- Static Site Generation (SSG)
- Server Side Rendering (SSR)

secara default semua nextjs merupakan Server Component, kecuali tambahin "use client". Karena bersifat server komponen, kita bisa melakukan request dari server dengan cara mendefinisikan fungsi halaman dengan `async` kemudian `await` untuk memanggil fungsi fetch, hasilnya browser akan langsung menerima HTML yang terender dengan data yang telah diambil.

## Kombinasi keduanya

Kita bisa melakukan trigger dari client seperti aksi `useEffect` dan menjalankan request di server.

### Client Component Sebagai Wrapper

Client Component akan menerima `children` berupa server component. Hal ini dikarenakan jika menulis server component dalam client komponen, akan otomatis menjadi client component

```jsx
import ClientDemo from '@/components/ClientDemo';
import RSCDemo from '@/components/RSCDemo';

export default function Home() {
  return (
    <main>
      <ClientDemo>
        <RSCDemo />
      </ClientDemo>
    </main>
  );
}
```

Dalam hal ini, `RSCDemo` dapat melakukan fetch data dari server side.

#### Layout

Khusus untuk layout, ketika kita menuliskan provider (misal untuk management state dengan context API), maka children (page utama) tidak akan menjadi `client` component.

### Server Actions

Buat file `action` berisi export fungsi `async` diatas file action tersebut gunakan `use server`. Kemudian import fungsi tersebut di `client` dan gunakan didalamnya.

#### Pendekatan Manual

```jsx
"use client"

import { useTransition } from "react"
import { submitData } from "@/app/actions"

export default function Form() {
  const [isPending, startTransition] = useTransition()

  const handleSubmit = (formData: FormData) => {
    startTransition(async () => {
      await submitData(formData)
    })
  }

  return (
    <form action={handleSubmit}>
      <button disabled={isPending}>
        {isPending ? "Loading..." : "Submit"}
      </button>
    </form>
  )
}
```

#### Pendekatan useTransitions

```jsx
// app/actions.ts
"use server"

export async function submitData(formData: FormData) {
  await new Promise((resolve) => setTimeout(resolve, 2000))
  return { success: true }
}

// app/components/form.ts
"use client"

import { useTransition } from "react"
import { submitData } from "@/app/actions"

export default function Form() {
  const [isPending, startTransition] = useTransition()

  const handleSubmit = (formData: FormData) => {
    startTransition(async () => {
      await submitData(formData)
    })
  }

  return (
    <form action={handleSubmit}>
      <button disabled={isPending}>
        {isPending ? "Loading..." : "Submit"}
      </button>
    </form>
  )
}

```

### Use() Hooks

`use()` hooks dapat digunakan untuk `client component` menunggu load data dari parent `server component`, sehingga tampilan tidak blank putih ketika load pertama.

```jsx
// app/pages/page.tsx
import { Suspense } from "react"
import ClientComponent from "./ClientComponent"

async function getData() {
  const res = await fetch("https://api.example.com/data")
  return res.json()
}

export default function Page() {
  const dataPromise = getData()

  return (
    <Suspense fallback={<p>Loading...</p>}>
      <ClientComponent data={dataPromise} />
    </Suspense>
  )
}

// app/pages/ClientComponent.tsx
"use client"

import { use } from "react"

export default function ClientComponent({ data }) {
  const resolvedData = use(data)

  return <div>{resolvedData.name}</div>
}
```

Atau bisa tambahkan `ErrorBoundary` untuk handler ketika error terjadi

```jsx
import fs from 'node:fs/promises';

import { Suspense } from 'react';

import UsePromiseDemo from '@/components/UsePromisesDemo';
import ErrorBoundary from '@/components/ErrorBoundary';

export default async function Home() {
  const fetchUsersPromise = new Promise((resolve, reject) =>
    setTimeout(async () => {
      const data = await fs.readFile('dummy-db.json', 'utf-8');
      const users = JSON.parse(data);
      // resolve(users);
      reject(new Error('Error!'));
    }, 2000)
  );

  return (
    <main>
      <ErrorBoundary fallback={<p>Something went wrong!</p>}>
        <Suspense fallback={<p>Loading users...</p>}>
          <UsePromiseDemo usersPromise={fetchUsersPromise} />
        </Suspense>
      </ErrorBoundary>
    </main>
  );
}
```

```jsx
'use client';

import { Component } from 'react';

export default class ErrorBoundary extends Component {
  constructor(props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, message: error.message };
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="error">
          <h2>An error occurred!</h2>
          <p>{this.state.message}</p>
          <p>{this.props.fallback}</p>
        </div>
      );
    }

    return this.props.children;
  }
}
```
